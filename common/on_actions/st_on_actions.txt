# Surplus Trade — On Actions
# Hooks into game lifecycle events to initialize variables and run the monthly pulse.
# Init event (st_events.0) also fires on monthly pulse to handle mid-game mod installation.

on_game_start_country = {
	events = {
		st_events.0
	}
}

on_monthly_pulse_country = {
	events = {
		st_events.0
		st_events.1
	}
}
