# Surplus Trade — Edicts
# System toggle (enable/disable) + 15 config cycling edicts (min/desired/max x 5 resources)
# All edicts are instant (length = 0) with no cost.

# ============================================================
# System Toggle
# ============================================================

st_enable_system = {
	length = 0
	icon = "GFX_edict_type_policy"

	potential = {
		st_is_player_country = yes
		NOT = { st_is_enabled = yes }
	}

	allow = {
		always = yes
	}

	effect = {
		hidden_effect = {
			set_variable = { which = st_system_enabled value = 1 }
		}
		country_event = { id = st_events.10 }
	}

	ai_will_do = {
		base = 0
	}
}

st_disable_system = {
	length = 0
	icon = "GFX_edict_type_policy"

	potential = {
		st_is_player_country = yes
		st_is_enabled = yes
	}

	allow = {
		always = yes
	}

	effect = {
		hidden_effect = {
			set_variable = { which = st_system_enabled value = 0 }
		}
		country_event = { id = st_events.11 }
	}

	ai_will_do = {
		base = 0
	}
}
